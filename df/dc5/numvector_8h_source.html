484 HD numvector<T, n> rotateLeft(size_t k) const
485 {
486 if (k > n)
487 throw;
488
489 if (k == n)
490 return *this;
491
492 numvector<T, n> res;
493
494 //for (size_t i = 0; i < n; ++i)
495 // res[i] = r[(i + k) % n];
496
497 for (size_t i = 0; i < n - k; ++i)
498 res[i] = this->data[i + k];
499 for (size_t i = n - k; i < n; ++i)
500 res[i] = this->data[i + k - n];
501 return res;
502 }//rotateLeft(...)
