481 HD numvector<T, n> rotateLeft(size_t k) const
482 {
483 if (k > n)
484 throw;
485
486 if (k == n)
487 return *this;
488
489 numvector<T, n> res;
490
491 //for (size_t i = 0; i < n; ++i)
492 // res[i] = r[(i + k) % n];
493
494 for (size_t i = 0; i < n - k; ++i)
495 res[i] = this->data[i + k];
496 for (size_t i = n - k; i < n; ++i)
497 res[i] = this->data[i + k - n];
498 return res;
499 }//rotateLeft(...)
