111void TimersGen::GenerateStatString(size_t stepNo, double curTime, size_t N)

115 std::stringstream ss;

124
125 for (const auto& k : timerLabelList)
126 {

129 tOther += (k == "Step" ? duration : -duration);

132
133 timestatFile << std::endl << ss.str();
