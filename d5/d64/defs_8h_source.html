337  {
338  std::string fname(name);
339 
340  size_t dec = 1;
341 
342  for (int i = 1; i < length; ++i)
343  {
344  dec *= 10;
345  if (number < dec)

367  inline void copyFile(const std::string& fileNameFrom, const std::string& fileNameTo)
368  {
369  std::string buf;

372  FILE *in, *out;

376 #pragma warning (push)
