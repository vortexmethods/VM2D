102 inline double norm(const std::vector<T>& b)
103 {
104 double norm = 0;
105 #ifndef OLD_OMP
106 #pragma omp simd reduction(+:norm)
107 #endif
108 for (size_t i = 0; i < b.size(); i++)
109 norm += (b[i] * b[i]);
110 return sqrt(norm);
111 }
