224 double VMlib::M4(double t)
225 {
226  double t2 = t*t;
227  double mt = t > 0 ? t : -t;
228 
229  return (mt > 2.0) ? 0.0 : \
230  (mt > 1.0) ? 0.5*sqr(2.0 - mt)*(1.0 - mt) : 1.0 - 2.5*t2 + 1.5*t2*mt;
