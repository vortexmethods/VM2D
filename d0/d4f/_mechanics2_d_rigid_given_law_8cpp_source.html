61 //Вычисление гидродинамической силы, действующей на профиль
62 void MechanicsRigidGivenLaw::GetHydroDynamForce()
63 {
64  W.getTimestat().timeGetHydroDynamForce.first += omp_get_wtime();
65 
66  const double& dt = W.getPassport().timeDiscretizationProperties.dt;
67 
68  hydroDynamForce = { 0.0, 0.0 };
69  hydroDynamMoment = 0.0;
70 
71  viscousForce = { 0.0, 0.0 };
72  viscousMoment = 0.0;
73 
74  Point2D hDFGam = { 0.0, 0.0 }; //гидродинамические силы, обусловленные присоед.завихренностью
75  Point2D hDFdelta = { 0.0, 0.0 }; //гидродинамические силы, обусловленные приростом завихренности
76  Point2D hDFQ = { 0.0, 0.0 }; //гидродинамические силы, обусловленные присоед.источниками
77 
78  double hDMGam = 0.0; //гидродинамический момент, обусловленный присоед.завихренностью
79  double hDMdelta = 0.0; //гидродинамический момент, обусловленный приростом завихренности
80  double hDMQ = 0.0; //гидродинамический момент, обусловленный присоед.источниками
81 
82 
83  for (size_t i = 0; i < afl.getNumberOfPanels(); ++i)
84  {
85  Point2D Vcm = VeloOfAirfoilRcm(W.getCurrentStep() * dt);
86  Point2D VcmOld = VeloOfAirfoilRcm((std::max(W.getCurrentStep(), (size_t)1) - 1) * dt);
87 
88  double Wcm = AngularVelocityOfAirfoil(W.getCurrentStep() * dt);
89  double WcmOld = AngularVelocityOfAirfoil((std::max(W.getCurrentStep(), (size_t)1) - 1) * dt);
90 
91  Point2D rK = 0.5 * (afl.getR(i + 1) + afl.getR(i));
92 
93  Point2D dr = rK - afl.rcm;
94 
95  Point2D velK = { -Wcm * dr[1], Wcm * dr[0] };
96  velK += Vcm;
97 
98  double gAtt = Wcm * (dr ^ afl.tau[i]) + (Vcm & afl.tau[i]);
99  double gAttOld = WcmOld * (dr ^ afl.tau[i]) + (VcmOld & afl.tau[i]);
100  double deltaGAtt = gAtt - gAttOld;
101 
102  double qAtt = Wcm * (dr ^ afl.nrm[i]) + (Vcm & afl.nrm[i]);
103 
105  double deltaK = boundary.sheets.freeVortexSheet(i, 0) * afl.len[i] - afl.gammaThrough[i] + deltaGAtt * afl.len[i];
106 
107  /*1*/
108  hDFdelta += deltaK * Point2D({ -dr[1], dr[0] });
109  hDMdelta += 0.5 * deltaK * dr.length2();
110 
111  /*2*/
112  hDFGam += 0.25 * (afl.getV(i) + afl.getV(i+1)).kcross() * gAtt * afl.len[i];
113  hDMGam += 0.25 * dr ^ (afl.getV(i) + afl.getV(i + 1)).kcross() * gAtt * afl.len[i];
114 
115  /*3*/
116  hDFQ -= 0.25 * (afl.getV(i) + afl.getV(i + 1)) * qAtt * afl.len[i];
117  hDMQ -= 0.25 * dr ^ (afl.getV(i) + afl.getV(i + 1)) * qAtt * afl.len[i];
118  }
119 
120  hydroDynamForce = hDFGam + hDFdelta * (1.0 / dt) + hDFQ;
121  hydroDynamMoment = hDMGam + hDMdelta / dt + hDMQ;
122 
123  if (W.getPassport().physicalProperties.nu > 0.0)
124  for (size_t i = 0; i < afl.getNumberOfPanels(); ++i)
125  {
126  Point2D rK = 0.5 * (afl.getR(i + 1) + afl.getR(i)) - afl.rcm;
127  viscousForce += afl.viscousStress[i] * afl.tau[i];
128  viscousMoment += (afl.viscousStress[i] * afl.tau[i]) & rK;
129  }
130 
131  W.getTimestat().timeGetHydroDynamForce.second += omp_get_wtime();
132 }// GetHydroDynamForce()

165 // Вычисление скоростей начал панелей
166 void MechanicsRigidGivenLaw::VeloOfAirfoilPanels(double currTime)
167 {
168  Point2D veloRcm = VeloOfAirfoilRcm(currTime);
169 
170  std::vector<Point2D> vel(afl.getNumberOfPanels(), { 0.0, 0.0 });
171  for (size_t i = 0; i < afl.getNumberOfPanels(); ++i)
172  {
173  vel[i][0] = (afl.getR(i) - afl.rcm).kcross()[0] * Wcm;
174  vel[i][1] = (afl.getR(i) - afl.rcm).kcross()[1] * Wcm;
175  vel[i] += veloRcm;
176  }
177 
178  afl.setV(vel);
179 }//VeloOfAirfoilPanels(...)

181 void MechanicsRigidGivenLaw::Move()
182 {
183  double t = W.getPassport().physicalProperties.getCurrTime();
184  double dt = W.getPassport().timeDiscretizationProperties.dt;
185 
186  //Point2D airfoilVelo = VeloOfAirfoilRcm(t);
187  Point2D aflRcmOld = afl.rcm;
188  double aflPhiOld = afl.phiAfl;
189  afl.Move(PositionOfAirfoilRcm(t + dt) - aflRcmOld);
190  afl.Rotate(AngleOfAirfoil(t + dt) - aflPhiOld);
191 
192  Vcm = VeloOfAirfoilRcm(t + dt);
193  Phi = AngleOfAirfoil(t + dt);
194  Wcm = AngularVelocityOfAirfoil(t + dt);
195 }//Move()
