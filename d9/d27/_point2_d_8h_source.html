186 template <char letter>
187 HD static bool cmp(const Point2D& ptA, const Point2D& ptB)
188 {
189 switch (letter)
190 {
191 case 0:
192 case 'x':
193 return (ptA[0] < ptB[0]);
194 case 1:
195 case 'y':
196 return (ptA[1] < ptB[1]);
197 case 'r':
198 return ((ptA[0] * ptA[0] + ptA[1] * ptA[1]) < (ptB[0] * ptB[0] + ptB[1] * ptB[1]));
199 default:
200 throw(-1);
201 return true;
202 }
203 }
204 };
